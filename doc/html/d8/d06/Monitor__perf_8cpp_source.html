00097 {
00098   if (perfds.find(tid) != perfds.end())
00099   {
00100     int ret;
00101     int fd = perfds[tid];
00102     ret = ioctl(fd, PERF_EVENT_IOC_RESET);
00103     if (ret == -1)
00104     {
00105       context.debug("perf: Could not reset counter!");
00106       REPORTV(Error::MONITOR, "reset", 
00107                                      "Could not reset perf for "+QString::number(tid));

00109     
00110     ret = startPerfCounter(fd);
00111     if (ret == -1)
00112     {
00113       context.debug("perf: Could not start counter!");

00117   }
00118   else
00119   {
00120     int newfd = createPerfCounter(tid);

00204 int Monitor_perf::createPerfCounter(int tid)
00205 {
00206   //Structure storing the preferences for perf
00207   struct perf_event_attr attr;
00208   memset(&attr, 0, sizeof(attr));
00209   
00210   pid_t pid = tid;
00211   //The counter is going to measure the performance of the
00212   //specified thread on all cores
00213   int cpu = -1;
00214   //only one counter per group will be used
00215   int group_fd = -1;
00216   //flags must be zero according to the perf documentation
00217   unsigned long flags = 0;
00218 
00219   //Setup the attr struct
00220   attr.type = PERF_TYPE_HARDWARE;
00221   attr.config = event_type;
00222   attr.disabled = 1;
00223 
00224   int fd = sys_perf_event_open(&attr, pid, cpu, group_fd, flags);
00225 
00226   if (fd == -EINVAL)
00227     return -1;
00228   else
00229     return fd;
00230 }
