00215 ProcessTree ObservedProcess::getProcessTree()
00216 {
00217     if (!isRunning())

00219 
00220     ProcessTree::autopin_tid_list tasks;
00221 
00222     ProcessTree::autopin_tid_list child_procs;
00223 
00224     std::deque<std::pair<int, int> > worklist;
00225 
00226     ProcessTree result(pid);
00227 
00228     CHECK_ERROR(tasks = service->getProcessThreads(pid), ProcessTree::empty)
00229     result.addProcessTaskList(pid, tasks);
00230 
00231     CHECK_ERROR(child_procs = service->getChildProcesses(pid), ProcessTree::empty);
00232 
00233     for (ProcessTree::autopin_tid_list::iterator it = child_procs.begin(); it != child_procs.end(); it++)
00234         worklist.push_back(std::pair<int, int> (pid, *it));
00235 
00236     while (!worklist.empty())
00237     {
00238         std::pair<int, int> work_item = worklist.front();
00239         worklist.pop_front();
00240 
00241         result.addChildProcess(work_item.first, work_item.second);
00242         CHECK_ERROR(tasks = service->getProcessThreads(work_item.second), ProcessTree::empty)
00243         result.addProcessTaskList(work_item.second, tasks);
00244 
00245         CHECK_ERROR(child_procs = service->getChildProcesses(work_item.second), ProcessTree::empty)
00246 
00247         for (ProcessTree::autopin_tid_list::iterator it = child_procs.begin(); it != child_procs.end(); it++)
00248             worklist.push_back(std::pair<int, int> (work_item.second, *it));
00249 
00250     }
00251 
00252     return result;
00253 }
