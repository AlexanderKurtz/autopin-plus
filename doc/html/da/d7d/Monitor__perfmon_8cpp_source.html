00309   if (ret != 0)
00310     REPORTV(Error::MONITOR,"perfmon_ptrace",

00318     REPORTV(Error::MONITOR,"perfmon_ptrace",
00319                                    "Perfmon: Task "+QString::number(tid)+" has already exited");
