00084 PinningHistory::pinning_list ControlStrategy::readPinnings(QString opt)

00086   PinningHistory::pinning_list result;

00088 
00089   if (config->configOptionExists(opt) <= 0)
00090     REPORT(Error::BAD_CONFIG,"option_missing","No pinning specified", result);
00091 
00092   pinnings = config->getConfigOptionList(opt);
00093 
00094   for (int i = 0; i < pinnings.size(); i++)
00095   {
00096     QStringList pinning = pinnings[i].split(':', QString::SkipEmptyParts, Qt::CaseSensitive);
00097     PinningHistory::autopin_pinning new_pinning;
00098     
00099     for (int j = 0; j < pinning.size(); j++)
00100     {
00101       bool ok;
00102       int cpu = pinning[j].toInt(&ok);
00103       if (ok && cpu >= 0)
00104         new_pinning.push_back(cpu);
00105       else
00106         REPORT(Error::BAD_CONFIG,"option_format",pinnings[i]+" is not a valid pinning", result);
00107     }
00108 
00109     result.push_back(new_pinning);
00110   }
00111   
00112   return result;
00113 }

00122   ProcessTree::autopin_tid_list task_set = ptree.getAllTasks();
00123

00131
